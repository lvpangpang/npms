## Button

### 默认配置
```jsx
import React from 'react'
import { Button } from '../src'

export default function Index() {
  return (
    <Button>吕肥肥</Button>
  )
}
```

### 跳转按钮
```jsx
import React from 'react'
import { Button } from '../src'

export default function Index() {
  return (
    <Button href="https://www.baidu.com">去百度</Button>
  )
}
```
